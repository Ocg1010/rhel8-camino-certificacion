INICIAR SSH-AGENT
ps aux | grep ssh-agent
eval $(ssh-agent)

CONECTARSE UTILIZANDO UNA LLAVE ESPECIFIVA
# ssh -i .ssh/new_pass rhel8cliente2
# ssh-copy-id -i .ssh/new_pass.pub rhel8cliente2


AGREGA LLAVES PARA AUTENTICACION
ssh-add .ssh/new_pass
PARA VER SI HAY LLAVES CARGADAS
ssh-add -l
ELIMINAR LLAVES
ssh-add -D

CUANDO CONECTAS A UN SERVIDOR POR PRIMERA VEZ, EL SERVIDOR TE MUESTRA EL FINGERPRINT DE SU CLAVE PÚBLICA. ASÍ PUEDES COMPARAR SI ES LA CLAVE CORRECTA Y EVITAR POSIBLES ATAQUES DE SUPLANTACIÓN.
ssh-keygen -lf ~/.ssh/id_rsa.pub
