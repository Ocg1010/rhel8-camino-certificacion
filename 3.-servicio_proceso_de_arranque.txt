systemctl list-units -t path

# Enumeras tipos de unidad (socket)
systemctl list-unit-files -t socket

2.- Enmascaramiento y desenmascaramiento
# Enmascaramiento (enlace a /dev/null, no se puede reiniciar ni iniciar)
systemctl mask firewalld.serivce

# Desenmascaramiento
systemctl unmask firewalld.service

3.- Seleccionando el objetivo (target) de arranque 
# Mostrar todos los targets
systemctl list-units --type target
# Cambiar de un target a otro (cambia el target aislando)
systemctl isolate multi-user.target
# Ver target por default
systemctl get-default 
# Establecer target predeterminado
systemctl set-default graphical.target

3.- Montar el sistema de archivos como lectura y escritura con "mount -o remount,rw /sysroot
Verificar con "mount" que ya tenga permisos de escritura (rw)
